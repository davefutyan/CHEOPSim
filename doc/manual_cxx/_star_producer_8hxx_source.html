185 inline void StarProducer::interpolateFlux(double* x_ref, double* y_ref, unsigned size_ref, double* x, double* y, unsigned size) {
186 
187  for (unsigned i=0; i<size; i++) {
188  for (unsigned j=1; j<size_ref; j++) {
189  if (x_ref[j]>x[i]) {
190  double mu = (x[i]-x_ref[j-1])/(x_ref[j]-x_ref[j-1]);
191  y[i] = y_ref[j-1]*(1.-mu)+y_ref[j]*mu;
192  break;
193  }
194  }
195  }

197 }
